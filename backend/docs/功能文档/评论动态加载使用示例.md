### 2. 在人设卡详情抽屉中使用评论区

```vue
<template>
  <el-drawer v-model="visible" size="60%">
    <el-tabs v-model="activeTab">
      <el-tab-pane label="详情" name="detail">

        <CommentSection
          :target-id="personaCard.id"
          :target-type="'persona'"
          :show-header="false"
        />
      </el-tab-pane>
    </el-tabs>
  </el-drawer>
</template>

<script setup lang="ts">
import { ref } from 'vue';
import CommentSection from '/@/components/CommentSection.vue';

const visible = ref(false);
const activeTab = ref('detail');
const personaCard = ref({});
const commentCount = ref(0);

const formatCount = (count: number) => {
  if (count >= 10000) {
    return `${(count / 10000).toFixed(1)}万`;
  } else if (count >= 1000) {

  }
  return count;
};
```

### 3. 自定义评论列表（不使用组件）

```vue
<template>
  <div class="custom-comment-list">
    <div v-for="comment in comments" :key="comment.id" class="comment-card">
      <div class="comment-header">
        <span>{{ comment.user_name }}</span>
        <span>{{ formatTime(comment.create_datetime) }}</span>
      </div>
      <div class="comment-content">{{ comment.content }}</div>
      <div class="comment-actions">
        <button @click="handleLike(comment.id)">
          点赞 ({{ comment.like_count }})
        </button>
      </div>
    </div>
    
    <!-- 加载更多 -->
    <button v-if="hasMore" @click="loadMore" :loading="loading">
      加载更多
    </button>
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue';
import { getComments, reactComment } from '/@/api/comment';

const props = defineProps({
  targetId: String,
  targetType: String
});

const comments = ref([]);
const page = ref(1);
const pageSize = ref(10);
const hasMore = ref(false);
const loading = ref(false);

const loadComments = async () => {
  loading.value = true;
  try {
    const response = await getComments(
      props.targetId,
      props.targetType,
      page.value,
      pageSize.value
    );
    
    if (response.code === 2000) {
      comments.value = [...comments.value, ...response.data];
      hasMore.value = response.has_more;
    }
  } finally {
    loading.value = false;
  }
};

const loadMore = () => {
  page.value += 1;
  loadComments();
};

const handleLike = async (commentId: string) => {
  await reactComment(commentId, 'like');
  // 刷新评论列表
  page.value = 1;
  comments.value = [];
  loadComments();
};

onMounted(() => {
  loadComments();
});
</script>
```

## 后端使用示例

### 1. 获取评论列表（视图层）

```python
from rest_framework.decorators import api_view
from rest_framework.response import Response
from mainotebook.content.services.comment_service import CommentService
from mainotebook.content.serializers.comment import CommentSerializer

@api_view(['GET'])
def get_comments(request):
    """获取评论列表（分页）"""
    target_id = request.query_params.get('target_id')
    target_type = request.query_params.get('target_type')
    page = int(request.query_params.get('page', 1))
    page_size = int(request.query_params.get('page_size', 10))
    
    # 使用服务层获取评论
    result = CommentService.get_comments_tree(
        target_id,
        target_type,
        page,
        page_size,
        user=request.user if request.user.is_authenticated else None
    )
    
    # 序列化评论
    serializer = CommentSerializer(result['comments'], many=True)
    
    return Response({
        'code': 2000,
        'msg': '获取成功',
        'data': serializer.data,
        'total': result['total'],
        'page': result['page'],
        'page_size': result['page_size'],
        'has_more': result['has_more']
    })
```

## 常见问题

### Q1: 如何调整每页显示的评论数量？

A: 修改 `pageSize` 参数：

```typescript
// 前端
const pageSize = ref(20);  // 改为 20 条

// 后端
page_size = int(request.query_params.get('page_size', 20))  # 默认 20 条
```

### Q2: 如何禁用个性化推荐，只使用热度排序？

A: 在调用 `get_comments_tree` 时不传 `user` 参数：

```python
result = CommentService.get_comments_tree(
    target_id,
    target_type,
    page,
    page_size,
    user=None  # 不传用户，使用热度排序
)
```

### Q3: 如何清除所有用户的缓存？

A: 使用 Django 管理命令：

```bash
# 清除所有用户兴趣缓存
python manage.py shell
>>> from django.core.cache import cache
>>> cache.delete_pattern("user_interests:*")

# 清除所有浏览历史缓存
>>> cache.delete_pattern("user_views:*")
```

### Q4: 如何监控缓存命中率？

A: 在 Redis 中查看统计信息：

```bash
redis-cli info stats | grep keyspace
```

或在代码中添加日志：

```python
import logging
logger = logging.getLogger(__name__)

cache_key = f"user_interests:{user_id}"
interests = cache.get(cache_key)
if interests:
    logger.info(f"缓存命中: {cache_key}")
else:
    logger.info(f"缓存未命中: {cache_key}")
```
